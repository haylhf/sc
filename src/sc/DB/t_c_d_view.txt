
SELECT V_t_c.*,
d.`name` AS department_name
FROM

(SELECT
tc.teacher_id,
t.department_id,
tc.id AS teacher_course_id,
tc.profession_id,
tc.course_id,
c.course_property,
c.course_type,
c.course_credit,
c.`name` AS course_name,
b.`name` AS building_name,
tc.building_id,
tc.course_weekday,
tc.course_start_time,
tc.course_max_people,
tc.remark
FROM
sc_teacher_course AS tc 
LEFT JOIN sc_course AS c ON c.id = tc.course_id
LEFT JOIN sc_building AS b ON b.id = tc.building_id
LEFT JOIN sc_teacher AS t ON t.id = tc.teacher_id) AS V_t_c

LEFT JOIN sc_department AS d ON d.id = V_t_c.department_id
ORDER BY V_t_c.teacher_course_id ASC
